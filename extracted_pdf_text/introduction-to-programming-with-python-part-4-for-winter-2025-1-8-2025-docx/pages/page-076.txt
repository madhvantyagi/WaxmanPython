def merge_sort(arr):
if len(arr) > 1:
# Finding the mid of the array
mid = len(arr) // 2
# Dividing the array into two halves
L = arr[:mid]
R = arr[mid:]
# Sorting the first half
merge_sort(L)
# Sorting the second half
merge_sort(R)
# Merging the sorted halves
merge(arr, L, R)
def merge(arr, L, R):
i = j = k = 0
# Copy data to temp arrays L[] and R[]
while i < len(L) and j < len(R):
if L[i] < R[j]:
arr[k] = L[i]
i += 1
else:
arr[k] = R[j]
j += 1
k += 1
# Checking if any element was left
while i < len(L):
arr[k] = L[i]
i += 1
k += 1
while j < len(R):
arr[k] = R[j]
j += 1
k += 1
arr = [38, 27, 43, 3, 9, 82, 10]
merge_sort(arr)
print("Sorted array is:", arr)
- The `merge_sort` function first checks if the array has more than one element. If it does, it finds the
midpoint and splits the array into two halves, `L` and `R`.
- It then recursively calls itself to sort `L` and `R` separately.
- Once `L` and `R` are sorted, it calls the `merge` function to combine these sorted halves into a single
sorted array.
